The Algorithm Book

A theoretical deep dive into 8 classic sorting algorithms. Learn how they work, their time complexities, and see the code implementation.

1. Bubble Sort

Best CaseO(n)
Average CaseO(n²)
Worst CaseO(n²)
SpaceO(1)

Bubble Sort is relatively simple. It works by repeatedly stepping through the list, comparing adjacent elements, and swapping them if they are in the wrong order. The pass through the list is repeated until the list is sorted. It is termed "bubble sort" because the largest elements "bubble" to the top (end) of the array.

function bubbleSort(array) {
  let n = array.length;
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - i - 1; j++) {
      if (array[j] > array[j + 1]) {
        // Swap elements
        let temp = array[j];
        array[j] = array[j + 1];
        array[j + 1] = temp;
      }
    }
  }
  return array;
}

2. Selection Sort

Best CaseO(n²)
Average CaseO(n²)
Worst CaseO(n²)
SpaceO(1)

Selection sort separates the array into two parts: the sublist of items already sorted, and the sublist of items remaining to be sorted that make up the rest of the list. Initially, the sorted sublist is empty in the selection sort algorithm. Selection sort scans the unsorted sublist for the smallest item and swaps it with the leftmost unsorted element.

function selectionSort(array) {
  let n = array.length;
  for (let i = 0; i < n - 1; i++) {
    let minIdx = i;
    for (let j = i + 1; j < n; j++) {
      if (array[j] < array[minIdx]) {
        minIdx = j;
      }
    }
    // Swap the found minimum element with the first element
    let temp = array[minIdx];
    array[minIdx] = array[i];
    array[i] = temp;
  }
  return array;
}

3. Insertion Sort

Best CaseO(n)
Average CaseO(n²)
Worst CaseO(n²)
SpaceO(1)

Insertion Sort builds the final sorted array one item at a time. It works similarly to the way you might sort playing cards in your hands. You pick one element and insert it into its correct position compared to the already sorted portion of the array.

function insertionSort(array) {
  let n = array.length;
  for (let i = 1; i < n; i++) {
    let key = array[i];
    let j = i - 1;
    // Move elements greater than key to one position ahead
    while (j >= 0 && array[j] > key) {
      array[j + 1] = array[j];
      j = j - 1;
    }
    array[j + 1] = key;
  }
  return array;
}

4. Merge Sort

Best CaseO(n log n)
Average CaseO(n log n)
Worst CaseO(n log n)
SpaceO(n)

Merge sort is a Divide and Conquer algorithm. It divides input array into two halves, calls itself for the two halves, and then it merges the two sorted halves. It requires extra memory to hold the divided sub-arrays, but is incredibly fast even on massive datasets.

function mergeSort(array) {
  if (array.length <= 1) return array;

  const mid = Math.floor(array.length / 2);
  const left = mergeSort(array.slice(0, mid));
  const right = mergeSort(array.slice(mid));

  return merge(left, right);
}

function merge(left, right) {
  let result = [], leftIndex = 0, rightIndex = 0;
  while (leftIndex < left.length && rightIndex < right.length) {
    if (left[leftIndex] < right[rightIndex]) {
      result.push(left[leftIndex]);
      leftIndex++;
    } else {
      result.push(right[rightIndex]);
      rightIndex++;
    }
  }
  return result.concat(left.slice(leftIndex)).concat(right.slice(rightIndex));
}

5. Quick Sort

Best CaseO(n log n)
Average CaseO(n log n)
Worst CaseO(n²)
SpaceO(log n)

Quick Sort also uses the divide-and-conquer strategy constraint. It selects a 'pivot' element from the array and partitions the other elements into two sub-arrays, according to whether they are less than or greater than the pivot. The sub-arrays are then sorted recursively.

function quickSort(array, low = 0, high = array.length - 1) {
  if (low < high) {
    let pi = partition(array, low, high);
    quickSort(array, low, pi - 1);
    quickSort(array, pi + 1, high);
  }
  return array;
}

function partition(array, low, high) {
  let pivot = array[high];
  let i = low - 1;
  for (let j = low; j < high; j++) {
    if (array[j] < pivot) {
      i++;
      let temp = array[i];
      array[i] = array[j];
      array[j] = temp;
    }
  }
  let temp = array[i + 1];
  array[i + 1] = array[high];
  array[high] = temp;
  return i + 1;
}

6. Heap Sort

Best CaseO(n log n)
Average CaseO(n log n)
Worst CaseO(n log n)
SpaceO(1)

Heap sort works by visualizing the array elements as a special kind of complete binary tree called a heap. Once the data structure is built into a Max-Heap, you know the largest element is at the root. You swap it with the last element, remove it from the heap, and then heapify the rest of the list again.

function heapSort(array) {
  let n = array.length;
  // Build max heap
  for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
    heapify(array, n, i);
  }
  // Extract elements from heap one by one
  for (let i = n - 1; i > 0; i--) {
    let temp = array[0];
    array[0] = array[i];
    array[i] = temp;
    heapify(array, i, 0);
  }
  return array;
}

function heapify(array, n, i) {
  let largest = i;
  let left = 2 * i + 1;
  let right = 2 * i + 2;

  if (left < n && array[left] > array[largest]) largest = left;
  if (right < n && array[right] > array[largest]) largest = right;

  if (largest != i) {
    let swap = array[i];
    array[i] = array[largest];
    array[largest] = swap;
    heapify(array, n, largest);
  }
}

7. Shell Sort

Best CaseO(n log n)
Average CaseO(n log² n)
Worst CaseO(n²)
SpaceO(1)

Shell sort is a highly efficient sorting algorithm and is based on insertion sort algorithm. This algorithm avoids large shifts as in case of insertion sort, if the smaller value is to the far right and has to be moved to the far left. It uses a sequence of gradually shrinking gap values to structure elements.

function shellSort(array) {
  let n = array.length;
  for (let gap = Math.floor(n / 2); gap > 0; gap = Math.floor(gap / 2)) {
    for (let i = gap; i < n; i += 1) {
      let temp = array[i];
      let j;
      for (j = i; j >= gap && array[j - gap] > temp; j -= gap) {
        array[j] = array[j - gap];
      }
      array[j] = temp;
    }
  }
  return array;
}

8. Cocktail (Shaker) Sort

Best CaseO(n)
Average CaseO(n²)
Worst CaseO(n²)
SpaceO(1)

Cocktail sort, also known as bidirectional bubble sort, cocktail shaker sort, shaker sort, ripple sort, or shuffle sort, is a variation of bubble sort that sorts in both directions on each pass through the list. It is marginally faster than a standard bubble sort because it deals with "turtles" (small elements near the end) more quickly.

function cocktailSort(array) {
  let isSwapped = true;
  let start = 0;
  let end = array.length;

  while (isSwapped == true) {
    isSwapped = false;
    for (let i = start; i < end - 1; ++i) {
      if (array[i] > array[i + 1]) {
        let temp = array[i];
        array[i] = array[i + 1];
        array[i + 1] = temp;
        isSwapped = true;
      }
    }
    if (isSwapped == false) break;
    
    isSwapped = false;
    end = end - 1;
    
    for (let i = end - 1; i >= start; i--) {
      if (array[i] > array[i + 1]) {
        let temp = array[i];
        array[i] = array[i + 1];
        array[i + 1] = temp;
        isSwapped = true;
      }
    }
    start = start + 1;
  }
  return array;
}
